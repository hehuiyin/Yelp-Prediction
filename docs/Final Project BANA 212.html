houserange=[]
for i,v in quantile.iterrows():
    for a,b in zillow2.iterrows():
        if v['state']==b['State']:
            if b['HousePrice']<=v['quan1']:
                houserange.append(1)
            elif (b['HousePrice'] > v['quan1']) & (b['HousePrice']<=v['quan2']):
                houserange.append(2)
            elif (b['HousePrice'] > v['quan2'] )& (b['HousePrice']<=v['quan3']):
                houserange.append(3)
            else:
                houserange.append(4)
